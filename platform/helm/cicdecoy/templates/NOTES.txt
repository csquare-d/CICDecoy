=== Deploy Your First Decoy ===

  cat <<EOF | kubectl apply -f -
  apiVersion: cicdecoy.io/v1alpha1
  kind: Decoy
  metadata:
    name: my-first-decoy
    namespace: decoys-production
  spec:
    service:
      type: ssh
      port: 2222
    fidelity:
      tier: 2
    authentication:
      mode: realistic
  EOF
